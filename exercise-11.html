# Run models and a model selection process to evaluate what variables best predict each of the two response variables, maxlife(mo) and AFR(mo), from the following predictors: gestation(mo), newborn(g), weaning(mo), weanmass(g), litters/year, and overall body mass(g)

##Predictors for model selection

predictor_cols <- c("gestation(mo)", "newborn(g)", "weaning(mo)",
                    "weanmass(g)", "litters/year", "mass(g)")

run_model_selection <- function(response, data) {
  response_clean <- gsub("`", "", response)
  predictor_clean <- c("gestation(mo)", "newborn(g)", "weaning(mo)", "weanmass(g)", "litters/year", "mass(g)")
  
  full_formula <- reformulate(
    termlabels = paste0("`", predictor_clean, "`"),
    response = paste0("`", response_clean, "`"))
  
  vars_bare <- c(
    response_clean, predictor_clean, "`")
  
  data_clean<- data %>%
    drop_na(any_of(vars_bare))
  cat("\nRows used for", response, ":", nrow(data_clean), "\n")
  
  options(na.action = "na.fail")
  full_model<- lm(full_formula, data = data_clean)
  
  dredge_table <- dredge(full_model, rank = "AICc")
  
  return(list(
    full_model = full_model,
    dredge_table = dredge_table,
    data_clean = data_clean
  ))
}
